## Node (fetch)

```ts
const API_BASE = process.env.API_BASE // e.g. https://lisa-taurine.tera.space
const API_KEY = process.env.API_KEY

// Trigger an agent
const trigger = await fetch(`${API_BASE}/v1/consumer/agents/trigger`, {
  method: 'POST',
  headers: { 'content-type': 'application/json', authorization: `Bearer ${API_KEY}` },
  body: JSON.stringify({ agent: 'glitter', prompt: 'go to google', stepLimit: 20 })
}).then(r => r.json())

// Poll the run
const run = await fetch(`${API_BASE}/v1/consumer/agents/runs/${trigger.runId}`, {
  headers: { authorization: `Bearer ${API_KEY}` }
}).then(r => r.json())

// Stop the browser (optional)
await fetch(`${API_BASE}/v1/consumer/session?sessionId=${trigger.taurineSessionId}`, {
  method: 'DELETE', headers: { authorization: `Bearer ${API_KEY}` }
})
```

## Browser API quick calls

```bash
# Start a browser
curl -X POST "$API_BASE/v1/consumer/session" -H "Authorization: Bearer $API_KEY" -H "content-type: application/json" -d '{}'

# List sessions
curl "$API_BASE/v1/consumer/session/list?page=1&pageSize=20" -H "Authorization: Bearer $API_KEY"

# Stop a session
curl -X DELETE "$API_BASE/v1/consumer/session?sessionId=$SESSION_ID" -H "Authorization: Bearer $API_KEY"
```
