  <Tab title="Start Browser Session">
    <CodeGroup>

```python Python
import os
import requests

API_BASE = os.getenv("API_BASE")
API_KEY = os.getenv("API_KEY")

resp = requests.post(
    API_BASE + "/v1/consumer/session",
    headers={
        "Authorization": f"Bearer {API_KEY}",
        "content-type": "application/json",
    },
    json={},
)
data = resp.json()
print("sessionId:", data.get("sessionId"))
print("cdpUrl:", data.get("cdpUrl"))
```

```typescript TypeScript
const API_BASE = process.env.API_BASE!;
const API_KEY = process.env.API_KEY!;

const res = await fetch(API_BASE + "/v1/consumer/session", {
  method: "POST",
  headers: {
    Authorization: "Bearer " + API_KEY,
    "content-type": "application/json",
  },
  body: JSON.stringify({}),
});
const data = await res.json();
console.log("sessionId:", data.sessionId);
console.log("cdpUrl:", data.cdpUrl);
```

```bash cURL
curl -X POST "$API_BASE/v1/consumer/session" \
  -H "Authorization: Bearer $API_KEY" \
  -H "content-type: application/json" \
  -d '{}'
```

  <Tab title="List / Stop Sessions">
    <CodeGroup>

```python Python
import os
import requests

API_BASE = os.getenv("API_BASE")
API_KEY = os.getenv("API_KEY")

# List sessions
resp = requests.get(
    API_BASE + "/v1/consumer/session/list?page=1&pageSize=20",
    headers={"Authorization": f"Bearer {API_KEY}"},
)
print(resp.json())

# Stop a session
SESSION_ID = os.getenv("SESSION_ID")
if SESSION_ID:
    stop = requests.delete(
        API_BASE + "/v1/consumer/session",
        headers={"Authorization": f"Bearer {API_KEY}"},
        params={"sessionId": SESSION_ID},
    )
    print("stopped:", stop.status_code)
```

```typescript TypeScript
const API_BASE = process.env.API_BASE!;
const API_KEY = process.env.API_KEY!;

// List sessions
const list = await fetch(
  API_BASE + "/v1/consumer/session/list?page=1&pageSize=20",
  {
    headers: { Authorization: "Bearer " + API_KEY },
  }
);
console.log(await list.json());

// Stop a session
const SESSION_ID = process.env.SESSION_ID;
if (SESSION_ID) {
  const stop = await fetch(
    API_BASE + "/v1/consumer/session?sessionId=" + SESSION_ID,
    {
      method: "DELETE",
      headers: { Authorization: "Bearer " + API_KEY },
    }
  );
  console.log("stopped", stop.status);
}
```

```bash cURL
# List sessions
curl "$API_BASE/v1/consumer/session/list?page=1&pageSize=20" \
  -H "Authorization: Bearer $API_KEY"

# Stop a session
curl -X DELETE "$API_BASE/v1/consumer/session?sessionId=$SESSION_ID" \
  -H "Authorization: Bearer $API_KEY"
```
