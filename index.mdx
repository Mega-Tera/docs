  <Tab title="Start Browser Session">
    <CodeGroup>

```bash cURL
curl -X POST "https://lisa-taurine.tera.space/v1/consumer/session" \
  -H "Authorization: Bearer $BROWSER_API_KEY" \
  -H "content-type: application/json" \
  -d '{}'
```

```python Python
import os
import requests

BROWSER_API_KEY = os.getenv("BROWSER_API_KEY")

resp = requests.post(
    "https://lisa-taurine.tera.space/v1/consumer/session",
    headers={
        "Authorization": f"Bearer {BROWSER_API_KEY}",
        "content-type": "application/json",
    },
    json={},
)
data = resp.json()
print("sessionId:", data.get("sessionId"))
```

```typescript TypeScript
const BROWSER_API_KEY = process.env.BROWSER_API_KEY!;

const res = await fetch("https://lisa-taurine.tera.space/v1/consumer/session", {
  method: "POST",
  headers: {
    Authorization: `Bearer ${BROWSER_API_KEY}`,
    "content-type": "application/json",
  },
  body: JSON.stringify({}),
});
const data = await res.json();
console.log("sessionId:", data.sessionId);
```

  <Tab title="Trigger Agent Task">
    <CodeGroup>

```bash cURL
curl -X POST "$API_BASE/v1/task/create" \
  -H "Authorization: Bearer $API_KEY" \
  -H "content-type: application/json" \
  -d '{"agent":"glitter","prompt":"What does example.com\'s homepage say?","mode":"text","stepLimit":10}'
```

```python Python
import os
import requests

API_BASE = os.getenv("API_BASE")
API_KEY = os.getenv("API_KEY")

resp = requests.post(
    API_BASE + "/v1/task/create",
    headers={
        "Authorization": f"Bearer {API_KEY}",
        "content-type": "application/json",
    },
    json={"agent": "glitter", "prompt": "What does example.com's homepage say?", "mode": "text", "stepLimit": 10},
)
print(resp.json())
```

```typescript TypeScript
const API_BASE = process.env.API_BASE!;
const API_KEY = process.env.API_KEY!;

const res = await fetch(API_BASE + "/v1/task/create", {
  method: "POST",
  headers: {
    Authorization: "Bearer " + API_KEY,
    "content-type": "application/json",
  },
  body: JSON.stringify({ agent: "glitter", prompt: "What does example.com's homepage say?", mode: "text", stepLimit: 10 }),
});
console.log(await res.json());
```

  <Tab title="Get / Stop Session">
    <CodeGroup>

```bash cURL
# Get a session status
curl "https://lisa-taurine.tera.space/v1/consumer/session?sessionId=$SESSION_ID" \
  -H "Authorization: Bearer $BROWSER_API_KEY"

# Stop a session
curl -X DELETE "https://lisa-taurine.tera.space/v1/consumer/session?sessionId=$SESSION_ID" \
  -H "Authorization: Bearer $BROWSER_API_KEY"
```

```python Python
import os
import requests

BROWSER_API_KEY = os.getenv("BROWSER_API_KEY")
# Get a session status
SESSION_ID = os.getenv("SESSION_ID")
if SESSION_ID:
    status = requests.get(
        "https://lisa-taurine.tera.space/v1/consumer/session",
        headers={"Authorization": f"Bearer {BROWSER_API_KEY}"},
        params={"sessionId": SESSION_ID},
    )
    print(status.json())

    # Stop a session
    stop = requests.delete(
        "https://lisa-taurine.tera.space/v1/consumer/session",
        headers={"Authorization": f"Bearer {BROWSER_API_KEY}"},
        params={"sessionId": SESSION_ID},
    )
    print("stopped:", stop.status_code)
```

```typescript TypeScript
const BROWSER_API_KEY = process.env.BROWSER_API_KEY!;

// Get a session status
const SESSION_ID = process.env.SESSION_ID!;
const status = await fetch(
  `https://lisa-taurine.tera.space/v1/consumer/session?sessionId=${encodeURIComponent(SESSION_ID)}`,
  { headers: { Authorization: `Bearer ${BROWSER_API_KEY}` } }
);
console.log(await status.json());

// Stop a session
const stop = await fetch(
  `https://lisa-taurine.tera.space/v1/consumer/session?sessionId=${encodeURIComponent(SESSION_ID)}`,
  {
    method: "DELETE",
    headers: { Authorization: `Bearer ${BROWSER_API_KEY}` },
  }
);
console.log("stopped", stop.status);
```

  <Tab title="Get Agent Task">
    <CodeGroup>

```bash cURL
curl -H "Authorization: Bearer $API_KEY" \
  "$API_BASE/v1/task/$TASK_ID"
```

```python Python
import os
import requests

API_BASE = os.getenv("API_BASE")
API_KEY = os.getenv("API_KEY")
TASK_ID = os.getenv("TASK_ID")

resp = requests.get(
    API_BASE + "/v1/task/" + str(TASK_ID),
    headers={"Authorization": f"Bearer {API_KEY}"},
)
print(resp.json())
```

```typescript TypeScript
const API_BASE = process.env.API_BASE!;
const API_KEY = process.env.API_KEY!;
const TASK_ID = process.env.TASK_ID!;

const res = await fetch(API_BASE + "/v1/task/" + TASK_ID, {
  headers: { Authorization: "Bearer " + API_KEY },
});
console.log(await res.json());
```
