    function fixLinks(){
      var nodes=document.querySelectorAll('a[href^="/"]');
      nodes.forEach(function(a){
        var href=a.getAttribute('href')||'';
        if(!href) return;
        if(href.startsWith('/docs')) return;

      });
    }
    if(document.readyState!=='loading') fixLinks();
    else document.addEventListener('DOMContentLoaded',fixLinks);

  <Tab title="Stop Session">
    <CodeGroup>

```bash cURL
# Stop a session
curl -X DELETE "https://browser-api.browser.cash/v1/consumer/session?sessionId=$SESSION_ID" \
  -H "Authorization: Bearer $BROWSER_API_KEY"
```

```python Python
import os
import requests

BROWSER_API_KEY = os.getenv("BROWSER_API_KEY")
SESSION_ID = os.getenv("SESSION_ID")

resp = requests.delete(
    "https://browser-api.browser.cash/v1/consumer/session",
    headers={"Authorization": f"Bearer {BROWSER_API_KEY}"},
    params={"sessionId": SESSION_ID},
)
print("stopped:", resp.status_code)
```

```typescript TypeScript
const BROWSER_API_KEY = process.env.BROWSER_API_KEY!;
const SESSION_ID = process.env.SESSION_ID!;

const res = await fetch(
  `https://browser-api.browser.cash/v1/consumer/session?sessionId=${encodeURIComponent(SESSION_ID)}`,
  { method: "DELETE", headers: { Authorization: `Bearer ${BROWSER_API_KEY}` } }
);
console.log("stopped", res.status);
```
