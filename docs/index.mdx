```python Python
import os
import requests

        "content-type": "application/json",
    },
    json={},
)

```

  <Tab title="Trigger Agent Task">
    <CodeGroup>

```bash cURL
curl -X POST "https://api.browser.cash/v1/agent/task" \
  -H "Authorization: Bearer $BROWSER_API_KEY" \
  -H "content-type: application/json" \
  -d '{"agent":"gemini","prompt":"What text does the homepage of example.com say?","mode":"text","stepLimit":10}'
```

```python Python
import os
import requests

BROWSER_API_KEY = os.getenv("BROWSER_API_KEY")

resp = requests.post(
    "https://api.browser.cash/v1/agent/task",
    headers={
        "Authorization": f"Bearer {BROWSER_API_KEY}",
        "content-type": "application/json",
    },
    json={"agent": "gemini", "prompt": "What text does the homepage of example.com say?", "mode": "text", "stepLimit": 10},
)
data = resp.json()
print("taskId:", data.get("taskId"))
```

import BrowsercashSDK from '@browsercash/sdk';

  "prompt": "What text does the homepage of example.com say?",
  "mode": "text",
  "stepLimit": 10
})

  <Tab title="Stop Session">
    <CodeGroup>

```bash cURL
curl -X DELETE "https://api.browser.cash/v1/browser/session?sessionId=$SESSION_ID" \
  -H "Authorization: Bearer $BROWSER_API_KEY"
```

```python Python
import os
import requests

BROWSER_API_KEY = os.getenv("BROWSER_API_KEY")
SESSION_ID = os.getenv("SESSION_ID")

resp = requests.delete(
    "https://api.browser.cash/v1/browser/session",
    headers={"Authorization": f"Bearer {BROWSER_API_KEY}"},
    params={"sessionId": SESSION_ID},
)
print("stopped:", resp.status_code)
```

```typescript TypeScript
import BrowsercashSDK from '@browsercash/sdk';

const client = new BrowsercashSDK({
  apiKey: process.env.BROWSER_API_KEY!,
});

const session = await client.browser.session.stop({ sessionId: process.env.SESSION_ID! });

console.log(session);
```
