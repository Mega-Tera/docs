    function fixLinks(){
      var nodes=document.querySelectorAll('a[href^="/"]');
      nodes.forEach(function(a){
        var href=a.getAttribute('href')||'';
        if(!href) return;
        if(href.startsWith('/docs')) return;
        if(href.startsWith('/_next')||href.startsWith('/mintlify-assets')||href.startsWith('/logo')||href.startsWith('/socket.io')) return;
        a.setAttribute('href','/docs'+href);
      });
    }
    if(document.readyState!=='loading') fixLinks();
    else document.addEventListener('DOMContentLoaded',fixLinks);
  })()}

| Category      | Browser API                      | Agent API                                          |
| :------------ | :------------------------------- | :------------------------------------------------- |
| Inputs        | Session Parameters                  | Questions; Autonomous Tasks                        |
| Outputs       | Chrome Browsers             | Deep Research; Completed Actions                   |
| Latency       | &lt;3s boot time                 | &lt;5s boot time                                   |
| Rate Limits   | 2,000 concurrent sessions        | 2,000 concurrent sessions                          |
| Session Limit | 1 hr                             | 1 hr (or a configurable step limit)                |
| Pricing       | $0.09 / hr (per browser session) | `$2` per million input tokens, `$8` per million output tokens.   |

Notes

- Token metering occurs after a run completes
- Browser API sessions are billed at a 1 minute minimum
